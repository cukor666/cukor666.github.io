sqlx介绍

sqlx可以认为是Go语言内置database/sql的超集，它在优秀的内置database/sql基础上提供了一组扩展。

这些扩展中除了大家常用来查询的Get(dest interface{}, ...) error和Select(dest interface{}, ...) error外还有很多其他强大的功能。

安装sqlx

go get github.com/jmoiron/sqlx

基本使用

使用起来和标准库database/sql用法类似。

导包

import (
	"fmt"
	_ "github.com/go-sql-driver/mysql"	// 连接数据库，对应的数据库驱动包一定要导入
	"github.com/jmoiron/sqlx"			// sqlx的依赖
)

初始化

// 原生的是：var db *sql.DB
var db *sqlx.DB

func initDB() (err error) {
	dsn := "root:cukor@tcp(localhost:3306)/students?charset=utf8mb4&parseTime=True"
	// 原生的是sql.Open，然后在后面需要调用Ping方法，而sqlx的Connect方法中已经自动调用了Ping，不需要我们自己调
	db, err = sqlx.Connect("mysql", dsn)
	if err != nil {
		fmt.Printf("connect DB failed, err:%v\n", err)
		return
	}
	db.SetMaxOpenConns(20) // 设置最大连接数，原生的也有
	db.SetMaxIdleConns(10) // 设置闲暇的现场池，原生的也有
	return
}

表的结构体

// 表对应的结构体，在sqlx中要把属性放出去，因为sqlx内部使用的反射，tag最好也设置
type user struct {
    ID   int    `db:"id"`
    Name string `db:"name"`
    Age  int    `db:"age"`
}

单行查询

sqlx给我们提供了一个Get方法，它比原生的QueryRow使用起来更加简单。

func queryOne() {
    sqlStr := `select id, name, age from user where id = ?`
    var u user                   // 这里可以不用初始化，因为在db.Get方法会自动申请内存
    err := db.Get(&u, sqlStr, 4) // 这里第一个参数必须传地址，切片等类型也一样，一定要传地址
    if err != nil {
       fmt.Printf("query failed, err: %v\n", err)
       return
    }
    fmt.Printf("id: %d\tname: %s\tage: %d\n", u.ID, u.Name, u.Age)
}
id: 4   name: Alice     age: 19

多行查询

sqlx给我们提供一个Select方法，它比原生的Query使用起来更加简单。

func queryMore() {
    sqlStr := `select id, name, age from user`
    var users []user                 // 因为是多行，所以需要使用切片类型来存储，但是这里并不需要申请内存，因为sqlx帮我们自动申请了
    err := db.Select(&users, sqlStr) // 这里第一个参数必须传地址
    if err != nil {
       fmt.Printf("select failed, err: %v\n", err)
       return
    }
    for _, u := range users {
       fmt.Println(u)
    }
}
{1 Catherine 23}
{2 Khalil 28}
{4 Alice 19}
{5 David 36}
{7 Ben 28}
{8 Jelly 15}

插入、更新、删除

sqlx的这几个操作和原生的基本一样。

func insertData() {
    sqlStr := `insert into user(name, age) values (?, ?)`
    result, err := db.Exec(sqlStr, "Ken", 43)
    if err != nil {
       fmt.Printf("exec failed, err: %v\n", err)
       return
    }
    affected, err := result.RowsAffected()
    if err != nil {
       fmt.Printf("get rows affected failed, err: %v\n", err)
       return
    }
    fmt.Printf("insert success, affected %d\n", affected)
}
func updateDemo() {
    sqlStr := `update user set age = 38 where id = ?`
    result, err := db.Exec(sqlStr, 9)
    if err != nil {
       fmt.Printf("exec failed, err: %v\n", err)
       return
    }
    affected, err := result.RowsAffected()
    if err != nil {
       fmt.Printf("get rows affected failed, err: %v\n", err)
       return
    }
    fmt.Printf("update success, affected %d\n", affected)
}
func deleteDemo() {
    sqlStr := `delete from user where id = ?`
    result, err := db.Exec(sqlStr, 7)
    if err != nil {
       fmt.Printf("exec failed, err: %v\n", err)
       return
    }
    affected, err := result.RowsAffected()
    if err != nil {
       fmt.Printf("get rows affected failed, err: %v\n", err)
       return
    }
    fmt.Printf("delete success, affected %d\n", affected)
}

NamedExec

DB.NamedExec方法用来绑定SQL语句与结构体或map中的同名字段。

这个操作就相当于让我们传参的时候更加的明确，因为如果参数多了，占位符?也多了，这样容易出现对应起来比较费劲。

但是这个也有个坏处：使用它会使得代码量更多了一点。

func insertUser() {
    sqlStr := `insert into user(name, age) values (:name, :age)`
    _, err := db.NamedExec(sqlStr, map[string]interface{}{
       "name": "Cukor",
       "age":  18,
    })
    if err != nil {
       fmt.Printf("insert failed, err: %v\n", err)
       return
    }
}

NamedQuery

与DB.NamedExec同理，这里是支持查询。

func queryName() {
    sqlStr := `select id, name, age from user where name = :name`
    // 返回的是*Rows
    // 使用map
    rows, err := db.NamedQuery(sqlStr, map[string]interface{}{
       "name": "Khalil",
    })
    if err != nil {
       fmt.Printf("query failed, err: %v\n", err)
       return
    }
    defer rows.Close()
    for rows.Next() {
       var u user
       //err := rows.Scan(&u)
       err := rows.StructScan(&u) // 因为u是结构体类型的，所以不能直接使用Scan
       if err != nil {
          fmt.Printf("StructScan failed, err: %v\n", err)
          continue
       }
       fmt.Println(u)
    }

    // 使用结构体

    u := user{
       Name: "Catherine",
    }

    rows, err = db.NamedQuery(sqlStr, u)   // 这里不能使用匿名结构体
    if err != nil {
       fmt.Printf("query failed, err: %v\n", err)
       return
    }
    defer rows.Close()
    for rows.Next() {
       var u user
       err := rows.StructScan(&u)
       if err != nil {
          fmt.Printf("StructScan failed, err: %v\n", err)
          continue
       }
       fmt.Println(u)
    }
}
