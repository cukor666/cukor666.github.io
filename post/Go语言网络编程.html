Go语言网络编程

把分布在不同地理区域的计算机与专门的外部设备用通信线路互连成一个规模大、功能强的网络系统，从而使众多的计算机可以方便地互相传递信息、共享硬件、软件、数据信息等资源。
设备之间在网络中进行数据的传输，发送/接收数据。

网络：至少两台以上的计算机之间的消息通信。

通信的两个重要要素：

  • IP：每台计算机都有自己的唯一标识，用于区分不同的计算机，以便计算机之间的通信。
  • PORT：计算机上的应用接口，用于区分单个计算机上的不同应用（软件）。一般1000以内的是系统的端口。

设备之间进行传输的时候，必须遵照一定的规则 —》通信协议：

通信协议相当于加密和解密的过程。只有都遵守了同一个协议，那么才能交流。否则计算机A使用加密算法1进行数据的加密，而计算机B使用加密算法2进行解密，那么两个加密算法不是同一类那B这边就无法解密，那么A、B两台计算机就无法通信。

客户端

Go提供了net包，用来做网络编程的。net包下有一个Dial函数，这个是用于拨号，即给服务器端发送消息通信。

package main

import (
	"fmt"
	"net"
)

func main() {
	fmt.Println("客户端启动。。。。")
	connection, err := net.Dial("tcp", "localhost:9090")
	if err != nil {
		fmt.Println("客户端连接失败。。。")
		panic(err)
	}
	fmt.Printf("客户端连接成功，connection：%v\n", connection)
}

服务器端

net包下有一个Listen函数，用于监听端口，查看是否有客户端连接。

package main

import (
	"fmt"
	"net"
)

func main() {
	fmt.Println("服务器端启动")
	listen, err := net.Listen("tcp", "localhost:9090")
	if err != nil {
		fmt.Println("服务器启动失败。。。")
	}
	for {
		c, err2 := listen.Accept()
		if err2 != nil {
			fmt.Println("等待客户端连接失败。。。", err2)
		} else {
			fmt.Println("等待客户端连接成功")
			fmt.Printf("connection:%v, 接收到客户端信息：%v\n", c, c.RemoteAddr().String())
		}
	}
}

package main

import (
	"bufio"
	"fmt"
	"net"
	"os"
)

func main() {
	fmt.Println("客户端启动。。。。")
	connection, err := net.Dial("tcp", "localhost:9090")
	if err != nil {
		fmt.Println("客户端连接失败。。。")
		panic(err)
	}
	fmt.Printf("客户端连接成功，connection：%v\n", connection)

	// 连接成功之后，使用客户端程序发送消息给服务器端
	// os.Stdin代表终端标准输入
	reader := bufio.NewReader(os.Stdin)
	// 从终端读取一行用户输入的信息：
	// 从终端读取一行字符串，所以使用\n来做分隔符
	s, err2 := reader.ReadString('\n')
	if err2 != nil {
		fmt.Println("终端输入失败，err:", err)
	}
	// 将数据s发送给服务器：
	n, err3 := connection.Write([]byte(s))
	if err3 != nil {
		fmt.Println("发送失败。。。")
	}
	fmt.Printf("终端数据通过客户端发送成功，一共发送了%d字节的数据，并退出", n)
}

服务器端：

package main

import (
	"fmt"
	"net"
)

func main() {
	fmt.Println("服务器端启动")
	listen, err := net.Listen("tcp", "localhost:9090")
	if err != nil {
		fmt.Println("服务器启动失败。。。")
		return
	}
	for {
		connection, err2 := listen.Accept()
		if err2 != nil {
			fmt.Println("等待客户端连接失败。。。", err2)
		} else {
			fmt.Println("等待客户端连接成功")
			fmt.Printf("connection:%v, 接收到客户端信息：%v\n", connection, connection.RemoteAddr().String())
		}
		// 准备一个协程处理客户端服务请求：
		// 不同的客户端的请求，连接connection不一样的
		go process(connection)
	}
}

func process(connection net.Conn) {
	// 连接用完一定要关闭：
	defer connection.Close()
	for {
		// 将读取的数据存放到切片中
		buf := make([]byte, 1024)

		// 从connection连接中读取数据
		n, err := connection.Read(buf)
		if err != nil {
			fmt.Printf("读取数据错误：%v\n", err)
			return
		}

		// 如果读取数据没有错误，则将读取内容打印在服务器端
		fmt.Println(string(buf[:n]))
	}
}
