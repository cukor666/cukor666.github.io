高精度计算

高精度算法的存在。在计算机中一般使用int、double等来存储数字，但是这样存储之后，当需要存储的数据大于类型所提供的空间时就会存在数据溢出的情况，导致算式的计算出现错误。而高精度算法就是用来解决这类问题的，高精度算法就是在计算大数据的时候仍然可以精准的计算出结果的算法。其核心就是使用了简单的加减乘除按数位计算，并且使用字符串来存储数据这样就可以将大数据保存下来，并且算出对应的结果。

因为本算法考虑到后续的使用，所以使用面向对象的方式将高精度算法实现，使用C++来实现算法。

封装一个ZString类，专门用来存储数据和计算数据的。

class ZString
{
public:
	enum Target {
		NX = -1, //表示负号
		X = 1    //表示正号
	};
	ZString(std::string strNumber, int target);
	ZString(std::string strNumber);
	~ZString();
	std::string& getIntNum();
	std::string& getDoubleNum();
	std::string getResult();
	void static setNumber(int* temp, std::string& str, bool flag = true);
	bool static numberStyle(int* temp, int length);
	std::string static toString(int* temp, int length, bool flag = false);
	void add(ZString other);
	void sub(ZString other);
	void mult(ZString other);
	void div(ZString other, int precision = 10);
protected:
	int* add(std::string& num1, std::string& num2);
	int* sub(std::string& big, std::string& small);
	std::string mult(std::string& num1, std::string& num2);
	std::string div(std::string& num1, std::string& num2);
private:
	bool isSwap(ZString& other);
	bool isZero(std::string& str);
	int compare(std::string& num1, std::string& num2);
	int addTarget(ZString& other, bool flag);
	int subTarget(ZString& other, bool flag);
	int multTarget(ZString& other);
	int divTarget(ZString& other);
private:
	std::string intNum;		//存储原始数据的整数部分
	std::string doubleNum;	//存储原始数据的小数部分
	int target;				//用来做符号标记，有正好和负号
};

一些杂项但又很必要的类成员函数

//辅助宏函数，使用宏函数的好处是减少压栈所消耗的时间和内存
#define MAX(a, b) ((a) > (b) ? (a) : (b))
#define CHAR2INT(x) ((x) - '0')
#define INT2CHAR(X) ((X) + '0')

//构造与析构
ZString::ZString(std::string strNumber, int target) {
	std::string::iterator it = strNumber.begin();
	this->target = target == ZString::Target::NX ? -1 : 1;
	char ch;
	bool flag = false;
	while (it != strNumber.end()) {	//遍历字符串以给成员变量初始化
		ch = *it;
		if (ch == '.') {
			flag = true;
			ch = *(++it);
		}
		if (flag)	doubleNum += ch;
		else		intNum += ch;
		it++;
	}
	if (doubleNum.empty()) doubleNum = "0";
}
//委托构造，主要是实现默认的数据是正数
ZString::ZString(std::string strNumber) : ZString(strNumber, ZString::Target::X) {}

ZString::~ZString() {}

std::string& ZString::getIntNum() { return intNum; }
std::string& ZString::getDoubleNum() { return doubleNum; }
std::string ZString::getResult() { 
    return target == Target::NX ? "-" + intNum + '.' + doubleNum : intNum + '.' + doubleNum;
}
//将字符串转换成整型数据存储
//flag为true则正序存储，否则逆序存储
void ZString::setNumber(int* temp, std::string& str, bool flag) {
	int i = 0;
	if (flag) {
		std::string::iterator it = str.begin();
		while (it != str.end())
			temp[i++] = CHAR2INT(*it++);
	}
	else {
		std::string::reverse_iterator it = str.rbegin();
		while (it != str.rend())
			temp[i++] = CHAR2INT(*it++);
	}
}
//进位返回true，不进位返回false
bool ZString::numberStyle(int* temp, int length) {
	for (int i = 0; i < length - 1; i++) {
		temp[i + 1] += temp[i] / 10;
		temp[i] %= 10;
	}
	return temp[length - 1] == 0 ? false : true;
}
//将整型数据（数组）转换为字符串存储
std::string ZString::toString(int* temp, int length, bool flag) {
	std::string result;
	if (flag)
		for (int i = 0; i < length; i++)
			result += INT2CHAR(temp[i]);
	else
		for (int i = length - 1; i >= 0; i--)
			result += INT2CHAR(temp[i]);
	return result;
}
//默认操作数1大于操作数2，如果相反则交换，该函数就是用来判定是否交换，用于确定算式结果的符号
bool ZString::isSwap(ZString& other) {
	bool flag = false;
	switch (compare(this->intNum, other.intNum))
	{
	case -1: flag = true; break;
	case 0:
		if (-1 == this->doubleNum.compare(other.doubleNum))
			flag = true;
		break;
	}
	return flag;
}
//字符串对应是数字是零则返回true否则返回false
bool ZString::isZero(std::string& str) {
	for (int i = 0; i < str.size(); i++)
		if (str[i] != '0') return false;
	return true;
}
//比较两个操作数的大小
int ZString::compare(std::string& num1, std::string& num2) {
	size_t len_1 = num1.size(), len_2 = num2.size();
	if (len_1 > len_2)		return 1;
	else if (len_1 < len_2)	return -1;
	else					return num1.compare(num2);
}
//确定加法运算的结果符号
int ZString::addTarget(ZString& other, bool flag) {
	if (this->target * other.target > 0) //同号
		return this->target;
	else {
		if (this->target == Target::NX) return flag ? 1 : -1;
		else return flag ? -1 : 1;
	}
}
//确定减法运算的结果符号
int ZString::subTarget(ZString& other, bool flag) {
	if (this->target * other.target < 0) //异号
		return this->target;
	else {
		if (this->target == Target::NX) return flag ? 1 : -1;
		else return flag ? -1 : 1;
	}
}
//确定乘法运算的结果符号
int ZString::multTarget(ZString& other) { return this->target * other.target < 0 ? -1 : 1; }
//确定除法运算的结果符号
int ZString::divTarget(ZString& other) { return this->target * other.target < 0 ? -1 : 1; }

高精度加法

//大体逻辑部分
void ZString::add(ZString other) {
	if (other.target == Target::NX) {
		other.target = Target::X;
		sub(other);
		return;
	}
	//符号部分
	this->target = addTarget(other, isSwap(other));
	//小数部分
	size_t len_1 = this->doubleNum.size(), len_2 = other.doubleNum.size();
	int len = int(MAX(len_1, len_2) + 1);
	if (len_1 > len_2) {
		size_t n = len_1 - len_2;
		while (n--) other.doubleNum.push_back('0');
	}
	else {
		size_t n = len_2 - len_1;
		while (n--) this->doubleNum.push_back('0');
	}
	int* result = add(this->doubleNum, other.doubleNum);
	if (numberStyle(result, len))
		intNum[intNum.size() - 1] += 1;
	--len;
	this->doubleNum = toString(result, len);
	delete[] result;
	//整数部分
	len = (int)MAX(this->intNum.size(), other.intNum.size() + 1);
	result = add(this->intNum, other.intNum);
	if (!numberStyle(result, len))
		--len;
	this->intNum = toString(result, len);
	delete[] result;
}
//实际运算
int* ZString::add(std::string& num1, std::string& num2) {
	int* result = new int[MAX(num1.size(), num2.size()) + 1]{ 0 }, i = 0;
	std::string::reverse_iterator p1 = num1.rbegin(), p2 = num2.rbegin();
	while (p1 != num1.rend() && p2 != num2.rend())
		result[i++] = CHAR2INT(*p1++) + CHAR2INT(*p2++);
	while (p1 != num1.rend())
		result[i++] = CHAR2INT(*p1++);
	while (p2 != num2.rend())
		result[i++] = CHAR2INT(*p2++);
	return result;
}

高精度减法

void ZString::sub(ZString other) {
	if (other.target == Target::NX) {
		other.target = Target::X;
		add(other);
		return;
	}
	//符号部分
	bool flag = isSwap(other);
	this->target = subTarget(other, flag);
	//先确定两个操作数的大小
	if (flag) {
		this->intNum.swap(other.intNum); this->doubleNum.swap(other.doubleNum);
	}
	//小数部分
	size_t len_1 = this->doubleNum.size(), len_2 = other.doubleNum.size();
	int len = int(MAX(len_1, len_2) + 1);
	if (len_1 > len_2) {
		size_t n = len_1 - len_2;
		while (n--) other.doubleNum.push_back('0');
	}
	else {
		size_t n = len_2 - len_1;
		while (n--) this->doubleNum.push_back('0');
	}
	if (doubleNum.compare(other.doubleNum) == -1) {
		doubleNum[0] += 10;
		intNum[intNum.size() - 1] -= 1;
	}
	len = (int)len_1;
	int* result = sub(this->doubleNum, other.doubleNum);
	doubleNum = toString(result, len);
	delete[] result;
	//整数部分
	len = (int)intNum.size();
	result = sub(this->intNum, other.intNum);
	if (result[len - 1] == 0) --len;
	intNum = toString(result, len);
	delete[] result;
}

int* ZString::sub(std::string& big, std::string& small) {
	int* result = new int[big.size()]{ 0 }, num, i = 0;
	std::string::reverse_iterator pBig = big.rbegin(), pSmall = small.rbegin();
	while (pSmall != small.rend()) {
		num = *pBig - *pSmall;
		if (num < 0) {
			result[i++] = num + 10; *(pBig + 1) -= 1;
		}
		else
			result[i++] = num;
		++pBig; ++pSmall;
	}
	while (pBig != big.rend()) result[i++] = CHAR2INT(*pBig++);
	return result;
}

高精度乘法

void ZString::mult(ZString other) {
	this->target = multTarget(other);
	int n = int(doubleNum.size() + other.doubleNum.size());
	std::string num1 = intNum + doubleNum;
	std::string num2 = other.intNum + other.doubleNum;
	std::string result = mult(num1, num2);
	intNum.assign(result.begin(), result.begin() + (result.size() - n));
	doubleNum.assign(result.begin() + (result.size() - n), result.end());
}

std::string ZString::mult(std::string& num1, std::string& num2) {
	std::string result;
	int* temp = new int[num1.size() + num2.size()]{ 0 }, i = 0, count = 0;
	std::string::reverse_iterator p1 = num1.rbegin(), p2;
	while (p1 != num1.rend()) {
		i = count++;
		p2 = num2.rbegin();
		while (p2 != num2.rend())
			temp[i++] += CHAR2INT(*p1) * CHAR2INT(*p2++);
		++p1;
	}
	int len = i;
	for (i = 0; i < len; i++) {
		temp[i + 1] += temp[i] / 10;
		temp[i] %= 10;
	}
	while (i) result += INT2CHAR(temp[--i]);
	delete[] temp;
	return result;
}

高精度除法

void ZString::div(ZString other, int precision) {
	this->target = divTarget(other);
	std::string temp;
	size_t n = other.doubleNum.size();		//移动的小数点位数
	int count = int(n - doubleNum.size());	//需要补零的个数
	other.intNum += other.doubleNum;	//表示小数点已经移动
	if (count < 0) {	//不需要补零
		intNum.append(doubleNum.begin(), doubleNum.begin() + n);
		doubleNum.assign(doubleNum.begin() + n, doubleNum.end());
	}
	else if (count == 0) {
		intNum.append(doubleNum.begin(), doubleNum.begin() + n);
		doubleNum = "0";
	}
	else {	//需要补count个零
		intNum.append(doubleNum.begin(), doubleNum.end());
		while (count--) intNum.push_back('0');
		doubleNum = "0";
	}
	temp = div(intNum, other.intNum);	//余数存放在intNum中，结果存放在temp中
	if (isZero(intNum)) intNum = doubleNum;
	else intNum += doubleNum;
	n = intNum.size();
	count = int(precision - n + 1);
	if (count > 0) while (count--) intNum.push_back('0');
	else if (count < 0) while (count++) intNum.pop_back();
	doubleNum = div(intNum, other.intNum);
	intNum = temp;
}

std::string ZString::div(std::string& num1, std::string& num2) {
	if (compare(num1, num2) == -1) return "0";
	std::string str(num2), result;	//补零之前先把原始数据保存
	int* temp = new int[num1.size()]{ 0 }, n, num, i;
	std::string::reverse_iterator pBig = num1.rbegin(), pSmall = num2.rbegin();
	while (compare(num1, num2) != -1)
		num2.push_back('0');
	num2.pop_back();
	while (compare(num2, str) != -1) {	//num2 >= str
		n = 0;
		while (compare(num1, num2) != -1) {	//num1 >= num2
			pBig = num1.rbegin(); pSmall = num2.rbegin(); i = 0;
			while (pSmall != num2.rend()) {
				num = *pBig - *pSmall;
				if (num < 0) {
					temp[i++] = num + 10; *(pBig + 1) -= 1;
				}
				else
					temp[i++] = num;
				++pBig; ++pSmall;
			}
			while (pBig != num1.rend()) temp[i++] = CHAR2INT(*pBig++);
			while (i > 1 && temp[i - 1] == 0) --i; num1 = toString(temp, i);
			++n;
		}
		result += INT2CHAR(n);
		num2.pop_back();
	}
	num2 = str;
	delete[] temp;
	return result;
}
