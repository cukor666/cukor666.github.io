函数

function add(num1: number, num2: number): number {
  return num1 + num2;
}

console.log(result);

// 匿名写法

  return num1 * num2;

函数参数默认值

const mul = (num1: number, num2: number = 6): number => {
  return num1 * num2;
};

result = mul(8);

默认值必须在后面，要不然调用时无法推断。

函数类型别名

接口

这TND就是结构体啊喂。

interface Person {
  name: string;
  age: number;
  addr: string | null;
}

let p: Person = {
  name: "David",
  age: 54,
  addr: null,
};

function showPerson(p: Person) {
  console.log("姓名：" + p.name);
  console.log("年龄：" + p.age);
  console.log("地址：" + p.addr);
}

showPerson(p)
姓名：David
年龄：54
地址：null

接口和类型 的区别 interface（接口）和 type（类型别名）的对比：

  • 相同点：都可以给对象指定类型
  • 不同点:
    • 接口，只能为对象指定类型。它可以继承。
    • 类型别名，不仅可以为对象指定类型，实际上可以为任意类型指定别名
    • 先有的 interface，后有的 type，推荐使用 type

接口继承

使用extends关键字进行继承操作。

interface Person {
  name: string;
  age: number;
  addr: string | null;
}

let p: Person = {
  name: "David",
  age: 54,
  addr: null,
};

interface Student extends Person {
  sno: string;
  classes: string;
}

function showPerson(p: Person) {
  console.log("姓名：" + p.name);
  console.log("年龄：" + p.age);
  console.log("地址：" + p.addr);
}

showPerson(p);

let stu: Student = {
  name: "Cukor",
  age: 18,
  addr: "柳州",
  sno: "1001",
  classes: "信计班",
};

console.log("-----------------------------");

// 多态
showPerson(stu);

function showStudent(s: Student) {
  showPerson(stu);
  console.log("学号：" + s.sno);
  console.log("班级：" + s.classes);
}
console.log("-----------------------------");

showStudent(stu);

权限控制

  • public：类的所有成员都可以被类的实例获取。 不写默认就是public
  • private：类成员只能在当前类中被访问。
  • protected：类成员在类以及子类中可以被访问
class Person {
  // 自己私有，子类不可以访问
  private name: string;
  private age: number;
  // 子类可访问
  protected money: number;
  constructor(name: string, age: number, money: number) {
    this.name = name;
    this.age = age;
    this.money = money;
    console.log("构造函数/方法会自动调用，在对象被创建的时候");
  }

  // 不写默认公有
  show() {
    console.log("姓名：" + this.name);
    console.log("年龄" + this.age);
    console.log("存款：" + this.money);
  }
}

let p: Person = new Person("Cukor", 18, 100);

// 继承Person，可以拿到Person的内容，当然受权限限制
class Student extends Person {
  private sno: string;
  private classes: string;
  constructor(
    name: string,
    age: number,
    money: number,
    sno: string,
    classes: string
  ) {
    super(name, age, money);
    this.sno = sno;
    this.classes = classes;
  }
  show(): void {
    super.show();
    console.log("学号：" + this.sno);
    console.log("班级：" + this.classes);
  }
  addMoney(val: number) {
    this.money += val;
  }
}

console.log("---------------------------------");

let s: Student = new Student("Jerry", 16, 200, "1002", "软件班");
s.addMoney(500);

泛型

function add<T>(a: T, b: T): T {
  if (typeof a === "number" && typeof b === "number") {
    return a + b as unknown as T;
  } else if (typeof a === "string" && typeof b === "string") {
    return a + b as unknown as T;
  } else {
    throw new Error("Unsupported types for addition or concatenation.");
  }
}
const result1 = add(5, 3); // 数字相加，结果为 8
const result2 = add("Hello, ", "World!"); // 字符串连接，结果为 "Hello, World!"

console.log(result1);
console.log(result2);

其中as是类型断言，就是先转为unknown再转到T，衔接得挺好，但是编译出来的js文件并没有更改。

多参数类型

只需要使用逗号分隔，加上去即可。

function print_info<T1, T2>(a: T1, b: T2) {
  console.log(a);
  console.log(b);
}

泛型约束

和Java的类似。

使用extends关键字限制上限。

function add<T extends number | string>(a: T, b: T): T {
  if (typeof a === "number" && typeof b === "number") {
    return (a + b) as unknown as T;
  } else if (typeof a === "string" && typeof b === "string") {
    return (a + b) as unknown as T;
  } else {
    throw new Error("Unsupported types for addition or concatenation.");
  }
}

console.log(add(2, 3));
console.log(add("Hello", "World"));
// 类型“boolean”的参数不能赋值给类型“string”|“number”的参数
