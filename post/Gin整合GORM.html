项目搭建

使用IDE简单创建一个Go的项目。

安装依赖

go get -u gorm.io/driver/mysql		# 数据库驱动，以MySQL为例
go get -u gorm.io/gorm				# GORM框架本身
go get -u github.com/gin-gonic/gin	 # Gin框架本身

项目结构

├─.idea
├─controller
├─dao
├─models
├─routers
└─service
  • controller：控制层，与前端交互，用于处理前端的请求
  • service：业务层，细分控制层逻辑
  • dao：数据访问层，和数据库打交道，访问数据库中的数据表
  • models：模型层，数据库表的结构体描述
  • routers：路由层，与前端交互，获取前端请求，然后把事务分发给控制层

还有一个main.go文件，这个是后端的入口。

新建数据库

GORM没有办法帮我们创建数据库，所以创建数据库这个操作需要我们自己完成。

create database sugar_rush;
use sugar_rush;

dao层

在dao目录下，创建一个dao.go文件，用于处理当前dao包下的全局配置。例如dsn，连接数据库等。

package dao

import (
    "fmt"
    "gorm.io/driver/mysql"
    "gorm.io/gorm"
)

// 全局配置，因为其他地方要使用到它
var db *gorm.DB

// init 函数会自动调用，在被main包导dao包的时候，用来处理初始化操作
func init() {
    // dsn := "username:password@tcp(localhost:3306)/databasename?charset=utf8mb4&parseTime=True&loc=Local"
    dsn := "root:cukor@tcp(localhost:3306)/sugar_rush?charset=utf8mb4&parseTime=True&loc=Local"
    open, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
    if err != nil {
       fmt.Printf("gorm open database failed, err: %v\n", err)
       panic(err)
    }
    db = open
}

// 使用db自动迁移创建表，因为在dao层并不知道要创建什么表，所以封装一个函数，提供给models层自己调用
// 注意的是，第一个参数虽然是空接口类型，但是只能传结构体类型
func CreateTable(obj interface{}, tableName string) {
    err := db.AutoMigrate(&obj)
    if err != nil {
       fmt.Printf("create table failed, table name: %s, err: %v", tableName, err)
    }
}

models层

GORM可以在models层面创建对应的数据表。

在models文件夹下，新建user.go文件

package models

import (
	"Gin-Gorm-Integration/dao"	// 创建完表之后不需要init里调用db的自动迁移了，所以这里记得去掉
	"gorm.io/gorm"
)

// User 数据库层面会创建一个users的表
type User struct {
	gorm.Model
	Username string `json:"username" gorm:"type:varchar(20)"`
	Password string `json:"password" gorm:"type:varchar(20)"`
	Age      uint   `json:"age" gorm:"type:uint;size:128"`
	Gender   string `json:"gender" gorm:"type:char(4)"`
	Addr     string `json:"addr" gorm:"type:varchar(255)"`
}

// 导包的时候会自动调用，dao.CreateTable要提前写好，创建了表之后就可以注释掉了
//func init() {
//	dao.CreateTable(User{}, "users")
//}

主函数先测试一下。

package main

import (
	_ "Gin-Gorm-Integration/dao"
	_ "Gin-Gorm-Integration/models"
	"fmt"
)

func main() {
	fmt.Printf("测试")
}

上面的代码，只导入包，但并不使用它们的其他内容，我们只需要执行它们的init函数。

然后运行，结果就是在sugar_rush数据库下多了一个users表。

desc users;

可以看到，GORM帮我们自动创建了，但是有些字段（age）的类型并不是我们所指定的那样。

这个时候可以直接在数据库层面修改它们的类型。

插入数据

dao层：

新建userDao.go：

package dao

import (
    "Gin-Gorm-Integration/models"
    "fmt"
)

type UserDao struct {
}

func (u *UserDao) Insert(user models.User) int64 {
    tx := db.Create(&user) // 这里要传地址
    if tx.Error != nil {
       fmt.Printf("insert failed, err: %v\n", tx.Error)
    }
    return tx.RowsAffected
}

main.go：

package main

import (
    "Gin-Gorm-Integration/dao"
    "Gin-Gorm-Integration/models"
    "fmt"
)

func main() {
    var uDao dao.UserDao
    user := models.User{
       Username: "Cukor",
       Password: "123456",
       Age:      18,
       Gender:   "男",
       Addr:     "柳州",
    }
    i := uDao.Insert(user)
    fmt.Printf("影响了%d条数据\n", i)
}
select * from users;
# 1,2023-10-19 14:35:50.906,2023-10-19 14:35:50.906,,Cukor,123456,18,男,柳州

service层

userService.go：

package service

import (
    "Gin-Gorm-Integration/dao"
    "Gin-Gorm-Integration/models"
    "fmt"
)

type UserService struct {
}

func (s *UserService) InsertUser(u models.User) int {
    var uDao dao.UserDao
    insert := uDao.Insert(u)
    fmt.Printf("insert success %d\n", insert)
    return int(insert)
}

func (s *UserService) SelectById(id int) (u models.User) {
    var uDao dao.UserDao
    u = uDao.SelectById(id)
    return
}

userDao.go：

原来基础上添加下面的函数：

func (u *UserDao) SelectById(id int) (result models.User) {
	tx := db.Where("id = ?", id).First(&result)
	if tx.Error != nil {
		fmt.Printf("select failed, id: %d, err: %v\n", id, tx.Error)
	}
	return
}

main.go：

var uService service.UserService
user := models.User{Username: "Alice", Password: "ILoveYou", Age: 18, Gender: "女", Addr: "南宁"}
uService.InsertUser(user)
