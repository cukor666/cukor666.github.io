JDBCдёҺMySQLж•°жҚ®еә“

JDBCеҹәжң¬жҰӮеҝө

JDBCжҳҜJavaж•°жҚ®еә“иҝһжҺҘпјҲJava Database Connectivityпјүзҡ„зј©еҶҷпјҢе®ғжҳҜдёҖеҘ—з”ЁдәҺжү§иЎҢSQLиҜӯеҸҘзҡ„Java APIгҖӮеә”з”ЁзЁӢеәҸеҸҜд»ҘйҖҡиҝҮиҝҷеҘ—APIиҝһжҺҘеҲ°е…ізі»ж•°жҚ®еә“пјҢе№¶дҪҝз”ЁSQLиҜӯеҸҘжқҘе®ҢжҲҗеҜ№ж•°жҚ®еә“дёӯж•°жҚ®зҡ„жҹҘиҜўгҖҒжӣҙж–°е’ҢеҲ йҷӨзӯүж“ҚдҪңгҖӮ

JDBCжҸҗдҫӣдәҶдёҖз§Қз»ҹдёҖзҡ„гҖҒеҹәдәҺJavaиҜӯиЁҖзҡ„е…ізі»ж•°жҚ®еә“зј–зЁӢжҺҘеҸЈи§„иҢғпјҢиҜҘи§„иҢғе…Ғи®ёдҪ жҠҠSQLиҜӯеҸҘдҪңдёәеҸӮж•°йҖҡиҝҮJDBCжҺҘеҸЈеҸ‘йҖҒз»ҷиҝңз«Ҝж•°жҚ®еә“пјҢиҝңз«Ҝж•°жҚ®еә“жҺҘеҸ—еҲ°дҪ зҡ„SQLеҗҺиҝӣиЎҢиҜӯжі•еҲҶжһҗгҖҒйӘҢиҜҒпјҢз„¶еҗҺжү§иЎҢгҖҒе“Қеә”гҖӮ

иҝһжҺҘMySQLж•°жҚ®еә“зҡ„еҮҶеӨҮ

иҰҒдҪҝз”ЁJavaиҝһжҺҘMySQLж•°жҚ®еә“пјҢдҪ йңҖиҰҒеҮҶеӨҮMySQLзҡ„й©ұеҠЁеҢ…гҖӮжңҖж–°зүҲжң¬зҡ„й©ұеҠЁеҢ…еҸҜд»Ҙд»ҺMySQLе®ҳзҪ‘дёӢиҪҪпјҢдёӢиҪҪең°еқҖдёәпјҡhttp://dev.mysql.com/downloads/connector/j/гҖӮеҰӮдёӢеӣҫжүҖзӨәпјҲеҸӮиҖғпјүгҖӮ

дёӢиҪҪе№¶и§ЈеҺӢеҗҺпјҢдҪ дјҡеҫ—еҲ°дёҖдёӘjarеә“ж–Үд»¶пјҢз„¶еҗҺеңЁеҜ№еә”зҡ„йЎ№зӣ®дёӯеҜје…ҘиҜҘеә“ж–Үд»¶еҚіеҸҜгҖӮ

дҪҝз”ЁMySQLй©ұеҠЁеҢ…

еҰӮжһңдҪ е·Із»ҸдёӢиҪҪеҘҪдәҶMySQLзҡ„й©ұеҠЁеҢ…пјҢйӮЈд№ҲдҪ йңҖиҰҒеңЁеҜ№еә”зҡ„JavaйЎ№зӣ®дёӯеҜје…ҘиҜҘеә“ж–Үд»¶гҖӮе…·дҪ“ж“ҚдҪңж–№жі•еҸ–еҶідәҺдҪ дҪҝз”Ёзҡ„ејҖеҸ‘е·Ҙе…·гҖӮ

Eclipse

дҫӢеҰӮпјҢеҰӮжһңдҪ дҪҝз”Ёзҡ„жҳҜEclipseпјҢйӮЈд№ҲдҪ еҸҜд»ҘеңЁйЎ№зӣ®дёӯеҸій”®еҚ•еҮ»пјҢйҖүжӢ© Build Path -> Configure Build Path -> Libraries -> Add External JARsпјҢз„¶еҗҺйҖүжӢ©дҪ дёӢиҪҪзҡ„MySQLй©ұеҠЁеҢ…жүҖеңЁзҡ„дҪҚзҪ®пјҢжңҖеҗҺзӮ№еҮ» Apply and Close еҚіеҸҜгҖӮ

IntelliJ IDEA

еҰӮжһңдҪ дҪҝз”Ёзҡ„жҳҜIntelliJ IDEAпјҢйӮЈд№ҲдҪ еҸҜд»ҘжҢүз…§д»ҘдёӢжӯҘйӘӨжқҘеҜје…ҘMySQLй©ұеҠЁеҢ…пјҡ

  1. жү“ејҖIDEAпјҢйҖүжӢ© File -> Project Structure -> ModulesгҖӮ
  2. йҖүдёӯеҸідҫ§зҡ„ + еҸ·пјҢзӮ№еҮ» JARs or directoriesгҖӮ
  3. йҖүжӢ©дҪ з”өи„‘дёӯзҡ„MySQLй©ұеҠЁеҢ…жүҖеңЁдҪҚзҪ®гҖӮ
  4. еҜје…ҘиҝҷдёӘjarеҢ…еҗҺпјҢзӮ№еҮ» ApplyпјҢеҶҚзӮ№еҮ» OK еҚіеҸҜгҖӮ

иҝҷж ·дҪ е°ұжҲҗеҠҹең°еңЁIntelliJ IDEAдёӯеҜје…ҘдәҶMySQLй©ұеҠЁеҢ…гҖӮ

зӨәдҫӢ

еҰӮжһңдҪ е·Із»ҸжҲҗеҠҹең°е°ҶMySQLй©ұеҠЁеҢ…еҜје…ҘеҲ°дҪ зҡ„йЎ№зӣ®дёӯпјҢйӮЈд№ҲдҪ еҸҜд»ҘдҪҝз”ЁJavaзҡ„JDBC APIжқҘиҝһжҺҘMySQLж•°жҚ®еә“е№¶жү§иЎҢSQLиҜӯеҸҘгҖӮ

дёӢйқўжҳҜдёҖдёӘз®ҖеҚ•зҡ„зӨәдҫӢпјҢжј”зӨәеҰӮдҪ•дҪҝз”ЁJavaиҝһжҺҘMySQLж•°жҚ®еә“е№¶жү§иЎҢжҹҘиҜўж“ҚдҪңпјҡ

package cn.cukor.jdbctest;

import java.sql.*;

public class Main {
    public static void main(String[] args) {
        // ж•°жҚ®еә“иҝһжҺҘдҝЎжҒҜпјҢж №жҚ®иҮӘе·ұзҡ„е®һйҷ…жғ…еҶөи®ҫзҪ®
        String url = "jdbc:mysql://localhost:3306/students";    // е®һйҷ…ж•°жҚ®еә“зҡ„ең°еқҖ
        String username = "root";        // з”ЁжҲ·еҗҚ
        String password = "123456";      // еҜҶз Ғ

        // еҠ иҪҪй©ұеҠЁпјҲд»ҘMySQLдёәдҫӢпјү
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }

        // иҝһжҺҘж•°жҚ®еә“
        try (Connection conn = DriverManager.getConnection(url, username, password)) {
            // еҲӣе»әStatementеҜ№иұЎ
            Statement stmt = conn.createStatement();

            // жү§иЎҢжҹҘиҜўж“ҚдҪң
            ResultSet rs = stmt.executeQuery("select * from stu");

            // еӨ„зҗҶжҹҘиҜўз»“жһң
            System.out.println("sno\tname\tgander\tbirthday");
            while (rs.next()) {
                int sno = rs.getInt("sno");
                String name = rs.getString("name");
                String gander = rs.getString("gander");
                Date birthday = rs.getDate("birthday"); // java.sqlдёӯзҡ„Date
                System.out.printf("%d\t%s\t%s\t%s\n", sno, name, gander, birthday);
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

еңЁдёҠйқўзҡ„д»Јз ҒдёӯпјҢжҲ‘д»¬йҰ–е…ҲеҠ иҪҪдәҶMySQLзҡ„й©ұеҠЁпјҢз„¶еҗҺдҪҝз”Ё DriverManager.getConnection() ж–№жі•жқҘиҝһжҺҘMySQLж•°жҚ®еә“гҖӮжҺҘзқҖпјҢжҲ‘д»¬еҲӣе»әдәҶдёҖдёӘ Statement еҜ№иұЎпјҢе№¶дҪҝз”Ёе®ғжқҘжү§иЎҢSQLжҹҘиҜўж“ҚдҪңгҖӮжңҖеҗҺпјҢжҲ‘д»¬еӨ„зҗҶжҹҘиҜўз»“жһңе№¶иҫ“еҮәз»“жһңгҖӮ

дҪ йңҖиҰҒж №жҚ®дҪ иҮӘе·ұзҡ„е®һйҷ…жғ…еҶөжқҘдҝ®ж”№дёҠйқўд»Јз Ғдёӯзҡ„ж•°жҚ®еә“иҝһжҺҘдҝЎжҒҜпјҢеҢ…жӢ¬ж•°жҚ®еә“URLгҖҒз”ЁжҲ·еҗҚе’ҢеҜҶз ҒгҖӮ

иҝҷж®өд»Јз ҒдҪҝз”ЁдәҶJava 7дёӯеј•е…Ҙзҡ„ try-with-resources иҜӯеҸҘгҖӮиҝҷз§ҚиҜӯеҸҘеҸҜд»ҘзЎ®дҝқеңЁзЁӢеәҸжү§иЎҢе®ҢжҜ•еҗҺпјҢиҮӘеҠЁе…ій—ӯеңЁжӢ¬еҸ·дёӯеЈ°жҳҺзҡ„иө„жәҗгҖӮ

еңЁдёҠйқўзҡ„зӨәдҫӢд»Јз ҒдёӯпјҢжҲ‘д»¬еңЁжӢ¬еҸ·дёӯеЈ°жҳҺдәҶдёҖдёӘ Connection еҜ№иұЎгҖӮиҝҷж„Ҹе‘ізқҖеҪ“зЁӢеәҸжү§иЎҢе®ҢжҜ•еҗҺпјҢиҝҷдёӘ Connection еҜ№иұЎдјҡиў«иҮӘеҠЁе…ій—ӯпјҢж— йңҖжҲ‘д»¬жүӢеҠЁи°ғз”Ё close() ж–№жі•жқҘе…ій—ӯе®ғгҖӮ

try-with-resources иҜӯеҸҘеҸҜд»Ҙи®©жҲ‘д»¬жӣҙж–№дҫҝең°з®ЎзҗҶиө„жәҗпјҢйҒҝе…Қеӣ дёәеҝҳи®°е…ій—ӯиө„жәҗиҖҢеҜјиҮҙзҡ„й—®йўҳгҖӮдҪ еҸҜд»ҘеңЁJavaе®ҳж–№ж–ҮжЎЈдёӯдәҶи§ЈжӣҙеӨҡе…ідәҺ try-with-resources иҜӯеҸҘзҡ„дҝЎжҒҜгҖӮ

Connectionзұ»

Connection зұ»жҳҜJavaзҡ„JDBC APIдёӯзҡ„дёҖдёӘжҺҘеҸЈпјҢе®ғиЎЁзӨәдёҖдёӘеҲ°ж•°жҚ®еә“зҡ„иҝһжҺҘгҖӮдҪ еҸҜд»ҘдҪҝз”Ё Connection еҜ№иұЎжқҘеҲӣе»ә StatementгҖҒPreparedStatement е’Ң CallableStatement еҜ№иұЎпјҢз”ЁдәҺжү§иЎҢSQLиҜӯеҸҘгҖӮ

жӯӨеӨ–пјҢConnection зұ»иҝҳжҸҗдҫӣдәҶдёҖдәӣж–№жі•жқҘз®ЎзҗҶдәӢеҠЎгҖӮдҫӢеҰӮпјҢдҪ еҸҜд»ҘдҪҝз”Ё setAutoCommit() ж–№жі•жқҘи®ҫзҪ®жҳҜеҗҰиҮӘеҠЁжҸҗдәӨдәӢеҠЎпјҢдҪҝз”Ё commit() ж–№жі•жқҘжҸҗдәӨдәӢеҠЎпјҢдҪҝз”Ё rollback() ж–№жі•жқҘеӣһж»ҡдәӢеҠЎгҖӮ

жҖ»д№ӢпјҢConnection зұ»жҳҜJavaзЁӢеәҸдёҺж•°жҚ®еә“д№Ӣй—ҙдәӨдә’зҡ„ж ёеҝғжҺҘеҸЈд№ӢдёҖгҖӮдҪ еҸҜд»ҘдҪҝз”Ёе®ғжқҘжү§иЎҢSQLиҜӯеҸҘгҖҒз®ЎзҗҶдәӢеҠЎзӯүж“ҚдҪңгҖӮ

Statementзұ»

Statement зұ»жҳҜJavaзҡ„JDBC APIдёӯзҡ„дёҖдёӘжҺҘеҸЈпјҢе®ғиЎЁзӨәдёҖдёӘз”ЁдәҺжү§иЎҢSQLиҜӯеҸҘзҡ„еҜ№иұЎгҖӮдҪ еҸҜд»ҘдҪҝз”Ё Statement еҜ№иұЎзҡ„ executeQuery() ж–№жі•жқҘжү§иЎҢжҹҘиҜўж“ҚдҪңпјҢдҪҝз”Ё executeUpdate() ж–№жі•жқҘжү§иЎҢжҸ’е…ҘгҖҒжӣҙж–°е’ҢеҲ йҷӨж“ҚдҪңгҖӮ

Statement еҜ№иұЎйҖҡеёёз”ЁдәҺжү§иЎҢйқҷжҖҒSQLиҜӯеҸҘпјҢеҚідёҚеёҰеҸӮж•°зҡ„SQLиҜӯеҸҘгҖӮеҰӮжһңдҪ йңҖиҰҒжү§иЎҢеёҰеҸӮж•°зҡ„SQLиҜӯеҸҘпјҢйӮЈд№ҲеҸҜд»ҘдҪҝз”Ё PreparedStatement зұ»гҖӮ

йңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢжҜҸдёӘ Statement еҜ№иұЎйғҪдёҺдёҖдёӘзү№е®ҡзҡ„ Connection еҜ№иұЎзӣёе…іиҒ”гҖӮдҪ еҸҜд»ҘдҪҝз”Ё Connection еҜ№иұЎзҡ„ createStatement() ж–№жі•жқҘеҲӣе»әдёҖдёӘж–°зҡ„ Statement еҜ№иұЎгҖӮ

ResuleSetзұ»

ResultSet зұ»жҳҜJavaзҡ„JDBC APIдёӯзҡ„дёҖдёӘжҺҘеҸЈпјҢе®ғиЎЁзӨәж•°жҚ®еә“жҹҘиҜўз»“жһңзҡ„ж•°жҚ®иЎЁгҖӮдҪ еҸҜд»ҘдҪҝз”Ё ResultSet еҜ№иұЎзҡ„ж–№жі•жқҘйҒҚеҺҶжҹҘиҜўз»“жһңе№¶иҺ·еҸ–ж•°жҚ®гҖӮ

дҫӢеҰӮпјҢдҪ еҸҜд»ҘдҪҝз”Ё next() ж–№жі•жқҘз§»еҠЁеҲ°дёӢдёҖиЎҢпјҢдҪҝз”Ё getString()гҖҒgetInt() зӯүж–№жі•жқҘиҺ·еҸ–еҪ“еүҚиЎҢдёӯзҡ„ж•°жҚ®гҖӮжӯӨеӨ–пјҢResultSet зұ»иҝҳжҸҗдҫӣдәҶдёҖдәӣж–№жі•жқҘжӣҙж–°жҹҘиҜўз»“жһңдёӯзҡ„ж•°жҚ®гҖӮ

йңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢжҜҸдёӘ ResultSet еҜ№иұЎйғҪдёҺдёҖдёӘзү№е®ҡзҡ„ Statement еҜ№иұЎзӣёе…іиҒ”гҖӮеҪ“дҪ дҪҝз”Ё Statement еҜ№иұЎзҡ„ executeQuery() ж–№жі•жқҘжү§иЎҢжҹҘиҜўж“ҚдҪңж—¶пјҢиҜҘж–№жі•дјҡиҝ”еӣһдёҖдёӘж–°зҡ„ ResultSet еҜ№иұЎпјҢз”ЁдәҺиЎЁзӨәжҹҘиҜўз»“жһңгҖӮ

SQLжҹҘиҜўиҜӯеҸҘеҜ№ж•°жҚ®еә“зҡ„жҹҘиҜўж“ҚдҪңе°Ҷиҝ”еӣһдёҖдёӘResultSetеҜ№иұЎпјҢResultSetеҜ№иұЎз”ұжҢүвҖңеҲ—вҖқз»„з»Үзҡ„ж•°жҚ®иЎҢжһ„жҲҗгҖӮ

еңЁиҺ·еҸ–з»“жһңйӣҶдёӯзҡ„з»“жһңж—¶пјҢеҸҜд»ҘдҪҝз”Ёж•°жҚ®иЎЁзҡ„еӯ—ж®өеҗҚд№ҹеҸҜд»ҘдҪҝз”Ёеӯ—ж®өжүҖеңЁеҲ—зҡ„еҲ—ж•°гҖӮ

жіЁж„Ҹпјҡж— и®әеӯ—ж®өжҳҜдҪ•з§ҚеұһжҖ§пјҢжҖ»еҸҜд»ҘдҪҝз”ЁgetString(int columnIndex)жҲ–getString(String columnName)ж–№жі•иҝ”еӣһеӯ—ж®өеҖјзҡ„дёІиЎЁзӨәгҖӮ

е…ій—ӯиҝһжҺҘ

йңҖиҰҒзү№еҲ«жіЁж„Ҹзҡ„жҳҜпјҢResultSetеҜ№иұЎе’Ңж•°жҚ®еә“иҝһжҺҘеҜ№иұЎе®һзҺ°дәҶзҙ§еҜҶзҡ„з»‘е®ҡпјҢдёҖж—ҰиҝһжҺҘеҜ№иұЎиў«е…ій—ӯпјҢResultSetеҜ№иұЎдёӯзҡ„ж•°жҚ®з«ӢеҲ»ж¶ҲеӨұгҖӮиҝҷе°ұж„Ҹе‘ізқҖеә”з”ЁзЁӢеәҸеңЁдҪҝз”ЁResultSetеҜ№иұЎдёӯзҡ„ж•°жҚ®ж—¶еҝ…йЎ»е§Ӣз»ҲдҝқжҢҒе’Ңж•°жҚ®еә“зҡ„иҝһжҺҘпјҢзӣҙеҲ°еә”з”ЁзЁӢеәҸResultSetеҜ№иұЎдёӯзҡ„ж•°жҚ®жҹҘзңӢе®ҢжҜ•гҖӮдҫӢеҰӮпјҢеҰӮжһңеңЁд»Јз Ғ

ResultSet rs = sql.executeQuery("select * from students");

д№ӢеҗҺз«ӢеҲ»е…ій—ӯиҝһжҺҘ

con.close();

йӮЈд№ҲзЁӢеәҸе°Ҷж— жі•иҺ·еҸ–rsдёӯзҡ„ж•°жҚ®гҖӮ

жҺ§еҲ¶жёёж Ү

з»“жһңйӣҶзҡ„жёёж Үзҡ„еҲқе§ӢдҪҚзҪ®еңЁз»“жһңйӣҶз¬¬дёҖиЎҢзҡ„еүҚйқўпјҢз»“жһңйӣҶи°ғз”Ёnext()ж–№жі•еҗ‘дёӢпјҲеҗҺпјүз§»еҠЁжёёж ҮпјҢз§»еҠЁжҲҗеҠҹж—¶иҝ”еӣһtrueпјҢеҗҰеҲҷиҝ”еӣһfalseгҖӮеҰӮжһңйңҖиҰҒеңЁз»“жһңйӣҶдёӯдёҠдёӢпјҲеүҚеҗҺпјүз§»еҠЁгҖҒжҳҫзӨәз»“жһңйӣҶдёӯжҹҗжқЎи®°еҪ•жҲ–йҡҸжңәжҳҫзӨәиӢҘе№ІжқЎи®°еҪ•пјҢеҝ…йЎ»иҝ”еӣһдёҖдёӘеҸҜж»ҡеҠЁзҡ„з»“жһңйӣҶгҖӮдёәдәҶеҫ—еҲ°дёҖдёӘеҸҜж»ҡеҠЁзҡ„з»“жһңйӣҶпјҢйңҖдҪҝз”ЁдёӢиҝ°ж–№жі•иҺ·еҫ—дёҖдёӘStatementеҜ№иұЎпјҡ

Statement stmt = con.createStatement(int type, int concurrency);

з„¶еҗҺж №жҚ®еҸӮж•°typeгҖҒconcurrencyзҡ„еҸ–еҖјжғ…еҶөпјҢstmtиҝ”еӣһзӣёеә”зұ»еһӢзҡ„з»“жһңйӣҶпјҡ

ResultSet rs = stmt.executeQuery(SQLиҜӯеҸҘ);

typeеҸ–еҖјеҶіе®ҡж»ҡеҠЁж–№ејҸпјҢеҸ–еҖјеҰӮдёӢпјҡ

  • ResultSet.TYPE_FORWORD_ONLYпјҡз»“жһңйӣҶзҡ„жёёж ҮеҸӘиғҪеҗ‘дёӢж»ҡеҠЁ
  • ResultSet.TYPE_SCROLL_INSENSITIVEпјҡз»“жһңйӣҶзҡ„жёёж ҮеҸҜд»ҘдёҠдёӢж»ҡеҠЁпјҢеҪ“ж•°жҚ®еә“еҸҳеҢ–ж—¶пјҢеҪ“еүҚз»“жһңйӣҶдёҚеҸҳ
  • ResultSet.TYPE_SCROLL_SENSITIVEпјҡиҝ”еӣһеҸҜж»ҡеҠЁзҡ„з»“жһңйӣҶпјҢеҪ“ж•°жҚ®еә“еҸҳеҢ–ж—¶пјҢеҪ“еүҚз»“жһңйӣҶеҗҢжӯҘж”№еҸҳгҖӮ
  • ResultSet.CONCUR_READ_ONLYпјҡдёҚиғҪз”Ёз»“жһңйӣҶжӣҙж–°ж•°жҚ®еә“дёӯзҡ„иЎЁ
  • ResultSet.CONCUR_UPDATABLEпјҡиғҪз”Ёз»“жһңйӣҶжӣҙж–°ж•°жҚ®еә“дёӯзҡ„иЎЁ

ж»ҡеҠЁжҹҘиҜўеҘҪеғҸз»Ҹеёёз”ЁеҲ°ResultSetзҡ„дёӢиҝ°ж–№жі•гҖӮ

  • previous()пјҡе°Ҷжёёж Үеҗ‘дёҠз§»еҠЁпјҢиҜҘж–№жі•иҝ”еӣһbooleanзұ»еһӢж•°жҚ®пјҢеҪ“з§»еҠЁеҲ°з»“жһңйӣҶзҡ„з¬¬дёҖиЎҢд№ӢеүҚж—¶иҝ”еӣһfalse
  • beforeFirst()пјҡе°Ҷжёёж Үз§»еҲ°з»“жһңйӣҶзҡ„еҲқе§ӢдҪҚзҪ®пјҢеҚіеңЁз¬¬дёҖиЎҢд№ӢеүҚ
  • afterLast()пјҡе°Ҷжёёж Үз§»еҠЁеҲ°з»“жһңйӣҶзҡ„жңҖеҗҺдёҖиЎҢд№ӢеҗҺ
  • first()пјҡе°Ҷжёёж Үз§»еҠЁеҲ°з»“жһңйӣҶзҡ„з¬¬дёҖиЎҢ
  • last()пјҡе°Ҷжёёж Үз§»еҠЁеҲ°з»“жһңйӣҶзҡ„жңҖеҗҺдёҖиЎҢ
  • isAfterLast()пјҡеҲӨж–ӯжёёж ҮжҳҜеҗҰеңЁжңҖеҗҺдёҖиЎҢд№ӢеҗҺ
  • isBeforeFirst()еҲӨж–ӯжёёж ҮжҳҜеҗҰеңЁз¬¬дёҖиЎҢд№ӢеүҚ
  • isFirst()пјҡеҲӨж–ӯжёёж ҮжҳҜеҗҰжҢҮеҗ‘з»“жһңйӣҶзҡ„з¬¬дёҖиЎҢ
  • isLast()пјҡеҲӨж–ӯжёёж ҮжҳҜеҗҰжҢҮеҗ‘з»“жһңйӣҶзҡ„жңҖеҗҺдёҖиЎҢ
  • getRow()пјҡеҫ—еҲ°еҪ“еүҚжёёж ҮжүҖжҢҮеҗ‘зҡ„иЎҢеҸ·пјҢиЎҢеҸ·д»Һ1ејҖе§ӢпјҢеҰӮжһңз»“жһңйӣҶжІЎжңүиЎҢпјҢиҝ”еӣһ0
  • absolute(int row)пјҡе°Ҷжёёж Үз§»еҲ°еҸӮж•°rowжҢҮе®ҡзҡ„иЎҢ

еҰӮжһңrowеҸ–еҖјдёәиҙҹпјҢе°ұжҳҜеҜјж•°зҡ„иЎҢж•°

GetDBConnection.javaпјҡ

package cn.cukor.one;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class GetDBConnection {
    public static Connection connectDB(String DBName, String id, String p) {
        Connection connection = null;
        String url = "jdbc:mysql://localhost:3306/" + DBName;
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }
        try {
            connection = DriverManager.getConnection(url, id, p);
        } catch (SQLException e) {
            e.printStackTrace();
        }

        return connection;
    }
}

package cn.cukor.one;

import java.sql.Connection;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.Random;

public class Main {
    public static void main(String[] args) {
        Connection connection;
        Statement sql;
        ResultSet rs;
        connection = GetDBConnection.connectDB("students", "root", "123456");
        if (connection == null) return;
        try {
            sql = connection.createStatement(ResultSet.TYPE_SCROLL_SENSITIVE, ResultSet.CONCUR_READ_ONLY);
            rs = sql.executeQuery("select * from stu");

            rs.last();  // з§»еҠЁжёёж ҮеҲ°жңҖеҗҺдёҖиЎҢ
            int max = rs.getRow();
            System.out.println("иЎЁе…ұжңү" + max + "жқЎи®°еҪ•пјҢйҡҸжңәжҠҪеҸ–дёӨжқЎи®°еҪ•");
            int[] a = {0, 1};
            Random random = new Random();
            a[0] = random.nextInt(2) + 1;   // жёёж Үд»Һ1ејҖе§Ӣи®Ўж•°пјҢеҰӮжһңеҮәзҺ°0е°ұдјҡжҠҘй”ҷ
            while (a[0] == a[1]) {
                a[1] = random.nextInt(2);
            }
            System.out.println("a[0] = " + a[0]);
            System.out.println("a[1] = " + a[1]);
            for (int i = 0; i < a.length; i++) {
                rs.absolute(a[i]);     // жёёж Үз§»еҠЁеҲ°з¬¬iиЎҢ
                String sno = rs.getString("sno");
                String name = rs.getString("name");
                System.out.println(sno + "\t" + name);
            }

            connection.close();
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}
